# i18n 対応設計書

作成日: 2026-02-26

## 1. 概要

Agent Teams Dashboard に国際化（i18n）対応を実装する。
画面表示および API エラーメッセージ、バックエンドログメッセージを多言語対応する。

## 2. 要件

### 2.1 対応言語

| 言語 | コード |
|------|--------|
| 日本語 | `ja` |
| 英語 | `en` |
| 中国語 | `zh` |

### 2.2 フォールバック言語

- **初回アクセス時**: ブラウザ設定を検出し、サポート言語にマッチしない場合は英語にフォールバック
- **翻訳キーなし時**: 指定言語 → 英語 → キー文字列

### 2.3 翻訳範囲

- フロントエンド: すべての UI 文字列、エラーメッセージ
- バックエンド: API エラーメッセージ、ログメッセージ

#### 言語セレクター

```typescript
// src/components/common/LanguageSelector.tsx
import { useTranslation } from 'react-i18next';
import { useEffect } from 'react';

const LANGUAGES = [
  { code: 'ja', label: '日本語' },
  { code: 'en', label: 'English' },
  { code: 'zh', label: '中文' },
];

export function LanguageSelector() {
  const { i18n } = useTranslation();

  useEffect(() => {
    document.documentElement.lang = i18n.language;
  }, [i18n.language]);

  return (
    <select
      value={i18n.language}
      onChange={(e) => i18n.changeLanguage(e.target.value)}
      aria-label={i18n.t('a11y.language_selector')}
    >
      {LANGUAGES.map((lang) => (
        <option key={lang.code} value={lang.code}>
          {lang.label}
        </option>
      ))}
    </select>
  );
}
```

#### API 呼び出し

```typescript
// src/lib/apiClient.ts
import i18n from '@/i18n';

export async function apiFetch<T>(url: string, options?: RequestInit): Promise<T> {
  const response = await fetch(url, {
    ...options,
    headers: {
      ...options?.headers,
      'Accept-Language': i18n.language,
      'Content-Type': 'application/json',
    },
  });

  if (!response.ok) {
    const error = await response.json();
    throw new Error(error.message);
  }

  return response.json();
}
```

### 5.2 バックエンド

#### I18nService クラス

```python
# app/services/i18n_service.py
import json
from pathlib import Path
from typing import Any

class I18nService:
    _instance = None
    _translations: dict[str, dict[str, dict]] = {}

    SUPPORTED_LANGUAGES = ["ja", "en", "zh"]
    DEFAULT_LANGUAGE = "en"

    def __new__(cls):
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._load_translations()
        return cls._instance

    def _load_translations(self):
        locales_dir = Path(__file__).parent.parent / "locales"

        for lang in self.SUPPORTED_LANGUAGES:
            self._translations[lang] = {}
            lang_dir = locales_dir / lang

            if not lang_dir.exists():
                continue

            for json_file in lang_dir.glob("*.json"):
                namespace = json_file.stem
                with open(json_file, "r", encoding="utf-8") as f:
                    self._translations[lang][namespace] = json.load(f)

    def t(self, key: str, lang: str = DEFAULT_LANGUAGE, **params: Any) -> str:
        # 言語フォールバック
        if lang not in self.SUPPORTED_LANGUAGES:
            lang = self.DEFAULT_LANGUAGE

        parts = key.split(".", 1)
        if len(parts) == 1:
            namespace, msg_key = "common", parts[0]
        else:
            namespace, msg_key = parts

        # 指定言語で取得
        translation = (
            self._translations.get(lang, {})
            .get(namespace, {})
            .get(msg_key)
        )

        # 英語にフォールバック
        if translation is None and lang != self.DEFAULT_LANGUAGE:
            translation = (
                self._translations.get(self.DEFAULT_LANGUAGE, {})
                .get(namespace, {})
                .get(msg_key)
            )

        # キー文字列を返す
        if translation is None:
            return key

        # パラメータ置換
        if params:
            translation = translation.format(**params)

        return translation


i18n = I18nService()
```

#### LanguageMiddleware

```python
# app/middleware/language.py
from fastapi import Request
from starlette.middleware.base import BaseHTTPMiddleware

from app.services.i18n_service import i18n

class LanguageMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        accept_language = request.headers.get("Accept-Language", i18n.DEFAULT_LANGUAGE)
        lang = accept_language.split(",")[0].split("-")[0].strip()

        if lang not in i18n.SUPPORTED_LANGUAGES:
            lang = i18n.DEFAULT_LANGUAGE

        request.state.language = lang

        return await call_next(request)
```
